function afficherEtoiles(moyenne) {
  var stars = document.querySelectorAll('.star');
  stars.forEach(function(star, index) {
    if (index < moyenne) {
      star.style.color = 'gold';
    } else {
      star.style.color = '#ccc';
    }
  });
}

var notes = [3, 4, 5]; // Exemple de notes
var somme = notes.reduce((a, b) => a + b, 0);
var moyenne = somme / notes.length;
afficherEtoiles(moyenne);

document.getElementById('averageRating').textContent = "Moyenne des notes : " + moyenne.toFixed(1) + " étoiles";

function afficherEtoiles(moyenne) {
  var stars = document.querySelectorAll('.star');
  stars.forEach(function(star, index) {
    if (index < Math.floor(moyenne)) {
      star.style.color = 'gold';
    } else if (index == Math.floor(moyenne) && moyenne % 1 !== 0) {
      star.style.color = 'gold';
      star.style.clipPath = 'polygon(0 0, 50% 0, 50% 100%, 0 100%)';
    } else {
      star.style.color = '#ccc';
    }
  });
}

var moyenne = 3.5; // Exemple de moyenne
afficherEtoiles(moyenne);

document.getElementById('averageRating').textContent = "Moyenne des notes : " + moyenne.toFixed(1) + " étoiles";
